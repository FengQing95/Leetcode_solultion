```python
# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def addTwoNumbers(self, l1: ListNode, l2: ListNode) -> ListNode:
        index = 1; num = 0
        while l1 or l2:
            if not l1:
                num += l2.val * index
                l2 = l2.next
            elif not l2:
                num += l1.val * index
                l1 = l1.next
            else:
                num += (l1.val + l2.val) * index
                l1 = l1.next
                l2 = l2.next
            index *= 10
        
        res = ListNode(); cur = res
        while num:
            curValue = num % 10
            cur.val = curValue
            num = num // 10
            if num:
                cur.next = ListNode()
                cur = cur.next
        
        return res
```
